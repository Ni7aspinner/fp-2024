<command> ::= "Add" | "Delete"
<cline> ::= <command> <plan>
<plan> ::= <weekDay> "(" <routine> ")" 
<routine> ::= <exercise> | <exercise> ", " <routine> 
<exercise> ::= <sor> <name> | "Superset[" <routine> "]"
<sor> ::= <digits> "x" <digits>
<name> ::= <name> <char> | <char>
<char> ::= "a" | ... | "z"
<digits> ::= "1" | "2" | "3" | "4" | "5" | "6" | "7" | "8" | "9" | "0"
<weekDay> ::= "Monday" | "Tuesday" | "Thursday" | "Wednesday" | "Friday" | "Saturday" | "Sunday"
